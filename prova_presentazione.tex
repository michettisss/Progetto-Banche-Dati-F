    % --- SLIDE 1: Recap Domanda di Ricerca ---
    \begin{frame}{Recap: Domanda di Ricerca}
        La crescita economica è sufficiente per ridurre la povertà multidimensionale oppure sono necessari interventi mirati sul capitale umano e le infrastrutture?
        \vspace{0.5cm}
        \begin{block}{Obiettivo del Modello}
            Identificare quali fattori (regressori) influenzano maggiormente l'indice di sviluppo umano (HDI) e la povertà nei paesi selezionati.
        \end{block}
    \end{frame}
    
    % --- SLIDE 2: Scelta dei Paesi ---
    \begin{frame}{Scelta dei Paesi: Senegal, Ghana, Nigeria e Sud Africa}
        La scelta è ricaduta su questi paesi per le seguenti ragioni:
        \begin{itemize}
            \item \textbf{Eterogeneità:} Diversi livelli di povertà multidimensionale e strutture economiche.
            \item \textbf{Rilevanza Regionale:} Rappresentano economie chiave dell'Africa Subsahariana con differenti trend demografici.
            \item \textbf{Politiche Pubbliche:} Approcci diversificati nella gestione delle risorse e dei servizi sociali.
            \item \textbf{Dati:} Disponibilità di serie storiche comparabili (2013--2021).
        \end{itemize}
    \end{frame}
    
    % --- SLIDE 3: Variabili del Modello ---
    \begin{frame}{Variabili: Dipendente e Regressori}
        \begin{itemize}
            \item \textbf{Variabile Dipendente:} HDI (Human Development Index) / MPI.
            \item \textbf{Regressori (Variabili Indipendenti):}
            \begin{itemize}
                \item \textbf{Economici:} PIL pro capite, FDI (\% PIL).
                \item \textbf{Infrastrutturali:} Accesso all'elettricità, Acqua potabile.
                \item \textbf{Demografici:} Fertilità, Popolazione totale.
                \item \textbf{Sanitari:} Tasso di mortalità (maschile e femminile).
            \end{itemize}
        \end{itemize}
    \end{frame}

    % --- SLIDE FINALE: Sviluppo del Modello ---
    \begin{frame}{Sviluppo del Modello: Prossimi Passi}
        Il modello verrà implementato seguendo queste fasi:
        \begin{enumerate}
            \item \textbf{Analisi di Correlazione:} Verificare il legame statistico tra HDI e i singoli regressori.
            \item \textbf{Regressione Multipla:} Stimare l'impatto combinato delle variabili indipendenti sulla povertà multidimensionale.
            \item \textbf{Test di Robustezza:} Controllare se i risultati reggono cambiando i paesi o i periodi temporali.
            \item \textbf{Interpretazione Politica:} Tradurre i coefficienti del modello in raccomandazioni per policy mirate.
        \end{enumerate}
    \end{frame}
