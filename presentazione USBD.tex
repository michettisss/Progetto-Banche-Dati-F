		La crescita economica è sufficiente per ridurre la povertà multidimensionale oppure sono necessari interventi mirati?

			\item Dataset utilizzato: \href{https://ourworldindata.org/grapher/multidimensional-poverty-index-mpi-hot}{Our World in Data}
		\end{itemize}

	%----------------------
	\begin{frame}{Domande descrittive}
		\begin{itemize}
			\item Evoluzione della povertà multidimensionale (2013--2021)
			\item Confronto tra Senegal, Ghana, Nigeria
			\item Quali dei fattori della multidimensionalità incidono di più? (MPI)
		\end{itemize}
	\end{frame}
	
	%----------------------
	\begin{frame}{Domande comparative}
		\begin{itemize}
			\item Differenze tra i paesi analizzati
			\item Povertà multidimensionale vs povertà monetaria
			\item Crescita economica $\rightarrow$ riduzione della povertà?

			\item Livello di istruzione $\rightarrow$ riduzione della povertà?
			\item È efficace investire nel capitale umano?
		\end{itemize}

			\item Dataset comparabili
		\end{itemize}
	\end{frame}
